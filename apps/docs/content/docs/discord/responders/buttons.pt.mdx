Responda um bot찾o definindo o tipo do `Responder` para bot찾o.

```ts title="command.ts"
const row = createRow(
    new ButtonBuilder({
        customId: "/confirm/button", 
        label: "Confirmar", 
        style: ButtonStyle.Success
    })
);
await interaction.reply({ 
    flags: ["Ephemeral"], 
    components: [row] 
});
```

```ts title="responder.ts"
createResponder({
    customId: "/confirm/button",
    // [!code word:Button]
    types: [ResponderType.Button], cache: "cached",
    async run(interaction) {
        interaction.update({ 
            flags: ["Ephemeral"], 
            content: "Confirmado", 
            components: [] 
        });
    },
});
```
